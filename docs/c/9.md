## 25.求一个一元二次方程的根
```c
#include <stdio.h>
#include <math.h>
int main(void)
{
	double a,b,c,d;
	printf("请输入一元二次方程的3个系数");
	scanf("%lf%lf%lf",&a,&b,&c,&d);
	if(fabs(a=0) <=1e-6)
	{
		printf("输入有误,程序结束");
		return 0;
	}
	d = b*b-4*a*c;
	if(d<0)
		printf("此方程无实数根");
	else  if(fabs(d) <=1e-6)
		printf("此方程有两个相同的实数根:%lf\n",-b/(2*a));
	else
		printf("此方程的两个实数根为:%lf,&lf\n",(-b+sqrt(d))/2*a,(-b-sqrt(d))/2*a);
	return 0;
}
```

## 27.switch语句
```c

{

			printf("上二楼\n");
```
